SLA calculation script (Business rule — before insert)
// Business Rule: Set SLA due date on record creation
// Table: u_operations_related | When: Before Insert

(function executeRule(current, previous) {
  var now = new GlideDateTime();
  var slaMins = { '1': 60, '2': 240, '3': 480, '4': 1440, '5': 2880 };
  var mins = slaMins[current.priority.toString()] || 480;

  // Calculate business-hours-aware due date
  var gdt = new GlideDateTime(now);
  gdt.addSeconds(mins * 60);
  current.u_sla_due = gdt;
  current.u_escalation_level = '0';
  current.u_sla_breached = false;
})(current, previous);
SLA monitoring — scheduled flow (runs every 15 min)
// Scheduled Script Execution: SLA Monitor
// Frequency: Every 15 minutes

var gr = new GlideRecord('u_operations_related');
gr.addQuery('state', 'IN', '1,2,3'); // Open, In Progress, On Hold
gr.addNullQuery('u_sla_breached').or('u_sla_breached', false);
gr.query();

while (gr.next()) {
  var due = new GlideDateTime(gr.u_sla_due);
  var now = new GlideDateTime();
  var total = due.getNumericValue() - new GlideDateTime(gr.sys_created_on).getNumericValue();
  var remaining = due.getNumericValue() - now.getNumericValue();
  var consumed = 1 - (remaining / total);

  if (consumed >= 1) {
    gr.u_sla_breached = true;
    gr.u_escalation_level = '3';
    // Trigger critical escalation flow
    sn_fd.FlowAPI.startFlow('x_abc.critical_escalation_flow', { ticket: gr.sys_id });
  } else if (consumed >= 0.9) {
    gr.u_escalation_level = '2';
    sn_fd.FlowAPI.startFlow('x_abc.sla_escalation_flow', { ticket: gr.sys_id, level: '2' });
  } else if (consumed >= 0.5) {
    if (gr.u_escalation_level.toString() === '0') {
      gr.u_escalation_level = '1';
      sn_fd.FlowAPI.startFlow('x_abc.sla_escalation_flow', { ticket: gr.sys_id, level: '1' });
    }
  }
  gr.update();
}

Escalation flow script
// Scheduled Flow: SLA Escalation Action
// Triggered by: SLA Monitor script at thresholds

function triggerEscalation(ticketSysId, level) {
  var gr = new GlideRecord('u_operations_related');
  if (!gr.get(ticketSysId)) return;

  var mgr = gr.assignment_group.manager.email.toString();
  var director = gs.getProperty('x_abc.ops.director_email');

  if (level === '1') {
    // Warning — notify group manager only
    sendAlert(mgr, gr, 'SLA Warning: 50% consumed', 'warning');

  } else if (level === '2') {
    // Escalation — bump priority, notify manager + director
    var newPri = Math.max(1, parseInt(gr.priority.toString()) - 1);
    gr.priority = newPri.toString();
    sendAlert(mgr, gr, 'SLA Alert: 90% consumed — escalating', 'high');
    sendAlert(director, gr, '[CC] SLA Escalation — ' + gr.number, 'high');

  } else if (level === '3') {
    // Breach — critical alert all managers
    gr.u_sla_breached = true;
    sendAlert(mgr, gr, 'SLA BREACHED — immediate action required', 'critical');
    sendAlert(director, gr, 'SLA BREACH — ' + gr.number, 'critical');
  }

  gr.u_escalation_level = level;
  gr.update();
}
